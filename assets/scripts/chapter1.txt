# [escena][Nombre]

# [cambiar_escena][NombreScene]

# [personaje][Nombre][emocion][feliz|triste|duda]

# [if][id][respuesta1][...accion...]
# [mostrar][imagen][textureKey]
# [imagen][key:AssetKey][x:..][y:..][scale:..][depth:..]
# [imagen][key:AssetKey][follow:Personaje][offsetX:..][offsetY:..][scale:..][depth:..]
# Nota: en escenas con paneo, la cámara puede estar desplazada (por ejemplo scrollY=800),
# así que posiciones absolutas con y bajas (ej. y:200) pueden quedar fuera de la vista.

# [esperar][ms]
# [pregunta_escena][es:Pregunta][correcta:2][op1:..][op2:..][op3:..]
# [fin_escena]
